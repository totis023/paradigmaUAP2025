import Html

foldl: List a -> (b -> a -> b) -> b -> b
foldl l fx inicio = 
  case l of 
    [] -> inicio
    head :: tail -> foldl tail fx (fx inicio head)

map: List a -> (a -> b) -> List b
map l fx = foldl l (acc a -> acc ++ (fx a)::[]) []
